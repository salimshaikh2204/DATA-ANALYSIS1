--round and trunc function in date

SELECT EMPLOYEE_ID, 
FIRST_NAME, 
HIRE_DATE, 
round(HIRE_DATE,'MONTH'),trunc(HIRE_DATE,'MONTH')
FROM EMPLOYEES
ORDER BY HIRE_DATE;

SELECT EMPLOYEE_ID, 
FIRST_NAME, 
HIRE_DATE, 
round(HIRE_DATE,'year'),trunc(HIRE_DATE,'year')
FROM EMPLOYEES
ORDER BY HIRE_DATE;
